import matplotlib.pyplot as plt
import pandas as pd

# csv file into df
data = pd.read_csv('outcomes.csv')

categories = data['Outcome']
single_counts = data['Single']
married_counts = data['Married']

plt.figure(figsize=(10, 6))

#bar graph for single
plt.bar(categories, single_counts, color='b', label='Single')

# bar graph for married
plt.bar(categories, married_counts, bottom=single_counts, color='r', label='Married')

plt.title('Educational Outcomes by Marital Status')
plt.xlabel('Outcome')
plt.ylabel('Count')
plt.legend()
plt.grid(True)
plt.show()
